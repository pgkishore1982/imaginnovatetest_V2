CREATE TABLE IF NOT EXISTS public.tbl_employees
(
    empid integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    firstname character varying(300) COLLATE pg_catalog."default",
    lastname character varying(300) COLLATE pg_catalog."default",
    email character varying(100) COLLATE pg_catalog."default",
    phonenumber character varying(25) COLLATE pg_catalog."default",
    doj timestamp without time zone,
    salary numeric,
    CONSTRAINT tbl_employees_pkey PRIMARY KEY (empid)
)
